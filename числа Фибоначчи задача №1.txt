// нумерация членов ряда начинается с "0" => f(0) = 0, f(1) = 1 и т.д.

#include <iostream>
using namespace std;
 
int f(int n)
{
    int a = 0;
    int b = 1;
    int c = 0;
    for (int i = 0; i < n; i++)
    {
        c = a;
        a = a + b;
        b = c;
    }
    return a;
}
 
int main()
{
    int n;
    cin >> n;
    cout << f(n);
    return 0;
}
