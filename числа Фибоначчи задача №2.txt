// нумерация членов ряда начинается с "0" => f(0) = 0, f(1) = 1 и т.д.

#include <iostream>
#include <cmath>
using namespace std;
 
int f(long n)
{
    int a = 0;
    int b = 1;
    int c = 0;
    for (long i = 0; i < n; i++)
    {
        c = a;
        a = (a + b) % 10;
        b = c % 10;
    }
    return a;
}
 
int main()
{
    long n;
    cin >> n;
    cout << f(n);
    return 0;
}
