#include <iostream>
using namespace std;

int main()
{
    long a, b, c;
    cin >> a >> b;
    if (a > b)
    {
        while (b != 0) 
        {
            c = a % b;
            a = b;
            b = c;
        }
        cout << a;
    } 
    else
    {
        while (a != 0) 
        {
            c = b % a;
            b = a;
            a = c;
        }
        cout << b;
    }
    
    return 0;
}
